./bin/cedar authorize \
    --policies tiny_sandboxes/sample6/policy.txt \
    --entities tiny_sandboxes/sample6/entity.json \
    --request-json tiny_sandboxes/sample6/request.json


/*
Validation:

Is policy.txt valid based on the schema schema.json
*/

./bin/cedar validate \
    --policies tiny_sandboxes/sample6/policy.txt \
    --schema tiny_sandboxes/sample6/schema.json

/*
Evaluate:

Evaluate a Cedar expression
*/

./bin/cedar evaluate  \
    --request-json tiny_sandboxes/sample6/request.json \
    --entities tiny_sandboxes/sample6/entity.json \
     "principal.account.age >= 17"
