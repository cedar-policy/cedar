./bin/cedar authorize \
    --policies tiny_sandboxes/sample4/policy.txt \

/*

*/

    --policies tiny_sandboxes/sample4/policy.txt \
    --schema tiny_sandboxes/sample4/schema.json

/*
Evaluate:

Evaluate a Cedar expression
*/

./bin/cedar evaluate  \
    --request-json tiny_sandboxes/sample4/request.json \
    --entities tiny_sandboxes/sample4/entity.json \
    "resource.owner == User::\"bob\""
