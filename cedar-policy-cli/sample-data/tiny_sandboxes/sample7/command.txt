/*
 This example shows how to use list, map and records in the context

 Can PhotoFlash::Data::User::"alice"  view PhotoFlash::Data::Photo::"VacationPhoto94.jpg"

 Decision: Allow
*/

cargo run  authorize \
    --policies tiny_sandboxes/sample7/policy.txt \
    --entities tiny_sandboxes/sample7/entity.json \
    --request-json tiny_sandboxes/sample7/request.json


/*
Validation:

Is policy.txt valid based on the schema schema.json
*/

cargo run  validate \
    --policies tiny_sandboxes/sample7/policy.txt \
    --schema tiny_sandboxes/sample7/schema.json

/*
Evaluate:

Evaluate a Cedar expression
*/

cargo run  evaluate  \
    --request-json tiny_sandboxes/sample7/request.json \
    --entities tiny_sandboxes/sample7/entity.json \
     "context.role.contains(\"admin\")"
